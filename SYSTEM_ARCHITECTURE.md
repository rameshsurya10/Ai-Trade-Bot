## Database Schema

```sql
-- Table 1: Price candles
CREATE TABLE candles (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    timestamp DATETIME NOT NULL,
    symbol VARCHAR(20) NOT NULL,
    open DECIMAL(20, 8) NOT NULL,
    high DECIMAL(20, 8) NOT NULL,
    low DECIMAL(20, 8) NOT NULL,
    close DECIMAL(20, 8) NOT NULL,
    volume DECIMAL(20, 8) NOT NULL,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,

    UNIQUE(timestamp, symbol)
);

-- Index for fast queries
CREATE INDEX idx_candles_timestamp ON candles(timestamp);
CREATE INDEX idx_candles_symbol ON candles(symbol);


-- Table 2: Trade history
CREATE TABLE trades (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    timestamp_entry DATETIME NOT NULL,
    timestamp_exit DATETIME,
    symbol VARCHAR(20) NOT NULL,
    direction VARCHAR(10) NOT NULL,  -- 'LONG' or 'SHORT'

    -- Prices
    entry_price DECIMAL(20, 8) NOT NULL,
    exit_price DECIMAL(20, 8),
    stop_loss DECIMAL(20, 8) NOT NULL,
    take_profit DECIMAL(20, 8) NOT NULL,

    -- Size
    position_size DECIMAL(20, 8) NOT NULL,
    position_value DECIMAL(20, 8) NOT NULL,

    -- Results
    pnl DECIMAL(20, 8),
    pnl_percent DECIMAL(10, 4),
    exit_reason VARCHAR(20),  -- 'STOP_LOSS', 'TAKE_PROFIT', 'SIGNAL', 'MANUAL'

    -- Model info
    prediction_confidence DECIMAL(5, 4),
    model_version VARCHAR(50),

    -- Status
    status VARCHAR(20) DEFAULT 'OPEN',  -- 'OPEN', 'CLOSED', 'CANCELLED'

    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP
);


-- Table 3: Predictions log
CREATE TABLE predictions (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    timestamp DATETIME NOT NULL,
    symbol VARCHAR(20) NOT NULL,

    -- Prediction
    prediction VARCHAR(10) NOT NULL,  -- 'UP' or 'DOWN'
    probability_up DECIMAL(5, 4) NOT NULL,
    probability_down DECIMAL(5, 4) NOT NULL,
    confidence DECIMAL(5, 4) NOT NULL,

    -- Actual result (filled later)
    actual_direction VARCHAR(10),
    actual_return DECIMAL(10, 6),
    was_correct BOOLEAN,

    -- Trade taken?
    trade_taken BOOLEAN DEFAULT FALSE,
    trade_id INTEGER,

    -- Model info
    model_version VARCHAR(50),
    features_hash VARCHAR(64),  -- To detect feature drift

    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,

    FOREIGN KEY (trade_id) REFERENCES trades(id)
);


-- Table 4: System state
CREATE TABLE system_state (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    key VARCHAR(50) UNIQUE NOT NULL,
    value TEXT NOT NULL,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP
);

-- Example entries:
-- INSERT INTO system_state (key, value) VALUES ('capital', '10000');
-- INSERT INTO system_state (key, value) VALUES ('daily_pnl', '127.50');
-- INSERT INTO system_state (key, value) VALUES ('bot_status', 'RUNNING');


-- Table 5: Daily summary
CREATE TABLE daily_summary (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    date DATE UNIQUE NOT NULL,

    -- Performance
    starting_capital DECIMAL(20, 8),
    ending_capital DECIMAL(20, 8),
    daily_pnl DECIMAL(20, 8),
    daily_pnl_percent DECIMAL(10, 4),

    -- Trades
    total_trades INTEGER DEFAULT 0,
    winning_trades INTEGER DEFAULT 0,
    losing_trades INTEGER DEFAULT 0,

    -- Predictions
    total_predictions INTEGER DEFAULT 0,
    correct_predictions INTEGER DEFAULT 0,

    created_at DATETIME DEFAULT CURRENT_TIMESTAMP
);
```
